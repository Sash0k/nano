57  	public void write(Object source, Writer out) throws WriterException, MappingException {
58  		try {
59  			// entry validation
60  			validate(source, out);
61  	
62  			XmlSerializer serializer = factory.newSerializer();
63  			if (format.isIndent()) {
64  				serializer.setFeature(IDENT_PROPERTY, true);
65  			}
66  			serializer.setOutput(out);
67  			serializer.startDocument(format.getEncoding(), null);
68  			
69  			MappingSchema ms = MappingSchema.fromObject(source);
70  			RootElementSchema res = ms.getRootElementSchema();
71  			String namespace = res.getNamespace();
72  			String xmlName = res.getXmlName();
73  			
74  			serializer.startTag(namespace, xmlName);
75  			this.writeObject(serializer, source, namespace);
76  			serializer.endTag(namespace, xmlName);
77  			
78  			serializer.endDocument();
79  		} catch (MappingException me) {
80  			throw me;
81  		} catch (IllegalArgumentException iae) {
82  			throw new WriterException("Entry validation failure", iae);
83  		} catch (Exception e) {
84  			throw new WriterException("Error to write/serialize object", e);
85  		}
86  	}

88  	private void validate(Object source, Writer out) {
89  		if (source == null) {
90  			throw new IllegalArgumentException("Can not write null instance!");
91  		}
92  		
93  		if (out == null) {
94  			throw new IllegalArgumentException("Writer is null!");
95  		}
96  		
97  		if (Transformer.isPrimitive(source.getClass())) {
98  			throw new IllegalArgumentException("Can not write primitive or enum type object, " +
99  					"only Nano bindable complex type object can be accepted!");
100 		}
101 	}

117 	private void writeObject(XmlSerializer serializer, Object source, String namespace) throws Exception {
118 		MappingSchema ms = MappingSchema.fromObject(source);
119 		
120 		// write xml attributes first
121 		writeAttributes(serializer, source, ms);
122 		
123 		// write xml value if any
124 		writeValue(serializer, source, ms);
125 		
126 		// write xml elements last
127 		writeElements(serializer, source, ms, namespace);
128 
129 	}

131 	private void writeAttributes(XmlSerializer serializer, Object source, MappingSchema ms) throws Exception {
132 		Map<String, AttributeSchema> field2AttributeSchemaMapping = ms.getField2AttributeSchemaMapping();
133 		for(String fieldName : field2AttributeSchemaMapping.keySet()) {
134 			AttributeSchema as = field2AttributeSchemaMapping.get(fieldName);
135 			Field field = as.getField();
136 			Object value = field.get(source);
137 			if (value != null) {
138 				String attValue = Transformer.write(value, field.getType());
139 				if (!StringUtil.isEmpty(attValue)) {
140 					serializer.attribute(null, as.getXmlName(), attValue);
141 				}
142 			}
143 		}
144 	}

146 	private void writeValue(XmlSerializer serializer, Object source, MappingSchema ms) throws Exception {
147 		ValueSchema vs = ms.getValueSchema();
148 		if (vs == null) return; // no ValueSchema, do nothing
149 		
150 		Field field = vs.getField();
151 		Object value = field.get(source);
152 		if (value != null) {
153 			String text = Transformer.write(value, field.getType());
154 			if (!StringUtil.isEmpty(text)) {
155 				if(vs.isData()) {
156 					serializer.cdsect(text);
157 				} else {
158 					serializer.text(text);
159 				}
160 			}
161 		}
162 	}

164 	private void writeElements(XmlSerializer serializer, Object source, MappingSchema ms, String namespace) throws Exception {
165 		Map<String, Object> field2SchemaMapping = ms.getField2SchemaMapping();
166 		for (String fieldName : field2SchemaMapping.keySet()) {
167 			Object schemaObj = field2SchemaMapping.get(fieldName);
168 			if (schemaObj instanceof ElementSchema) {
169 				ElementSchema es = (ElementSchema)schemaObj;
170 				Field field = es.getField();
171 				Object value = field.get(source);
172 				if (value != null) {
173 					if (es.isList()) {
174 						this.writeElementList(serializer, value, es, namespace);
175 					} else {
176 						this.writeElement(serializer, value, es, namespace);
177 					}
178 				}
179 			}
180 		}
181 	}

183 	private void writeElementList(XmlSerializer serializer, Object source, ElementSchema es, String namespace) throws Exception {
184 		for(Object value : (List<?>)source) {
185 			this.writeElement(serializer, value, es, namespace);
186 		}
187 	}

189 	private void writeElement(XmlSerializer serializer, Object source, ElementSchema es, String namespace) throws Exception {
190 		Class<?> type = null;
191 		if (es.isList()) {
192 			type = es.getParameterizedType();
193 		} else {
194 			type = es.getField().getType();
195 		}
196 		
197 		if (source == null) return; // do nothing
198 		
199 		String xmlName = es.getXmlName();
200 		
201 		// primitives
202 		if(Transformer.isPrimitive(type)) {
203 			String value = Transformer.write(source, type);
204 			if(StringUtil.isEmpty(value)) return;
205 			
206 			serializer.startTag(namespace, xmlName);
207 			if(es.isData()) {
208 				serializer.cdsect(value);
209 			} else {
210 				serializer.text(value);
211 			}
212 			serializer.endTag(namespace, xmlName);
213 			
214 			return;
215 		}
216 		
217 		// object
218 		serializer.startTag(namespace, xmlName);
219 		this.writeObject(serializer, source, namespace);
220 		serializer.endTag(namespace, xmlName);
221 	}
