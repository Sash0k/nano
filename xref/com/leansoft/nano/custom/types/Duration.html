45      /**
46       * Constructs Duration from a String in an xsd:duration format - PnYnMnDTnHnMnS.
47       *
48       * @param duration String
49       * @throws IllegalArgumentException if the string doesn't parse correctly.
50       */
51      public Duration(String duration) throws IllegalArgumentException {
52          int position = 1;
53          int timePosition = duration.indexOf("T");
54  
55          // P is required but P by itself is invalid
56          if (duration.indexOf("P") == -1 || duration.equals("P")) {
57              throw new IllegalArgumentException();
58              // Messages.getMessage("badDuration"));
59          }
60  
61          // if present, time cannot be empty
62          if (duration.lastIndexOf("T") == duration.length() - 1) {
63              throw new IllegalArgumentException();
64  //                    Messages.getMessage("badDuration"));
65          }
66  
67          // check the sign
68          if (duration.startsWith("-")) {
69              isNegative = true;
70              position++;
71          }
72  
73          // parse time part
74          if (timePosition != -1) {
75              parseTime(duration.substring(timePosition + 1));
76          } else {
77              timePosition = duration.length();
78          }
79  
80          // parse date part
81          if (position != timePosition) {
82              parseDate(duration.substring(position, timePosition));
83          }
84      }

101         this.seconds = calendar.get(Calendar.SECOND);

103         if (years == 0 && months == 0 && days == 0 && hours == 0 &&

110     /**
111      * This method parses the time portion of a String that represents xsd:duration - nHnMnS.
112      *
113      * @param time
114      * @throws IllegalArgumentException if time does not match pattern
115      */
116     public void parseTime(String time) throws IllegalArgumentException {
117         if (time.length() == 0 || time.indexOf("-") != -1) {
118             throw new IllegalArgumentException();
119             //Messages.getMessage("badTimeDuration"));
120         }
121 
122         // check if time ends with either H, M, or S
123         if (!time.endsWith("H") && !time.endsWith("M") && !time.endsWith("S")) {
124             throw new IllegalArgumentException();
125             //Messages.getMessage("badTimeDuration"));
126         }
127 
128         try {
129             // parse string and extract hours, minutes, and seconds
130             int start = 0;
131 
132             // Hours
133             int end = time.indexOf("H");
134             // if there is H in a string but there is no value for hours,
135             // throw an exception
136             if (start == end) {
137                 throw new IllegalArgumentException();
138                 //Messages.getMessage("badTimeDuration"));
139             }
140             if (end != -1) {
141                 hours = Integer.parseInt(time.substring(0, end));
142                 start = end + 1;
143             }
144 
145             // Minutes
146             end = time.indexOf("M");
147             // if there is M in a string but there is no value for hours,
148             // throw an exception
149             if (start == end) {
150                 throw new IllegalArgumentException();
151 //                        Messages.getMessage("badTimeDuration"));
152             }
153 
154             if (end != -1) {
155                 minutes = Integer.parseInt(time.substring(start, end));
156                 start = end + 1;
157             }
158 
159             // Seconds
160             end = time.indexOf("S");
161             // if there is S in a string but there is no value for hours,
162             // throw an exception
163             if (start == end) {
164                 throw new IllegalArgumentException();
165                 //Messages.getMessage("badTimeDuration"));
166             }
167 
168             if (end != -1) {
169                 setSeconds(Double.parseDouble(time.substring(start, end)));
170             }
171         } catch (NumberFormatException e) {
172             throw new IllegalArgumentException();
173             //Messages.getMessage("badTimeDuration"));
174         }
175     }

337     /** This returns the xml representation of an xsd:duration object. */
338     public String toString() {
339         StringBuffer duration = new StringBuffer();
340 
341         duration.append("P");
342 
343         if (years != 0) {
344             duration.append(years).append("Y");
345         }
346         if (months != 0) {
347             duration.append(months).append("M");
348         }
349         if (days != 0) {
350             duration.append(days).append("D");
351         }
352         if (hours != 0 || minutes != 0 || seconds != 0.0) {
353             duration.append("T");
354 
355             if (hours != 0) {
356                 duration.append(hours).append("H");
357 
358             }
359             if (minutes != 0) {
360                 duration.append(minutes).append("M");
361 
362             }
363             if (seconds != 0) {
364                 if (seconds == (int)seconds) {
365                     duration.append((int)seconds).append("S");
366                 } else {
367                     duration.append(seconds).append("S");
368                 }
369             }
370         }
371 
372         if (duration.length() == 1) {
373             duration.append("T0S");
374         }
375 
376         if (isNegative) {
377             duration.insert(0, "-");
378         }
379 
380         return duration.toString();
381     }
